Platonic Alphabet Assignment

words = ["alfa", "bravo", "charlie", "delta", "echo", "foxtrot", "golf", "hotel", "india", "juliett", "kilo",
"lima", "mike", "november", "oscar", "papa", "quebec", "romeo", "sierra", "tango", "uniform", "victor", "whiskey", "xray", "yankee", "zulu"]

inp = input().lower()
output = ""

for letter in inp:
    for word in words:
        if letter == word[0]:
            output += word + " "

print(inp + " -> \n" + output)
